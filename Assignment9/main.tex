\begin{align}
    \omega_O &= \frac{1}{\sqrt{LC}}\\

\end{align}
At resonant frequency, 
\begin{align}
    X_c &= \frac{1}{\omega C}\\
    &= 2 \Omega\\
     \frac{V_o}{V_i} &= \frac{X_c}{R} \\
     &= \frac{2}{R}\\
     \abs{T(s)} &= 20 \log\frac{V_o(s)}{V_i(s)}\\
     26 &= 20 \log\frac{2}{R}\\
     R &= 0.1 \Omega
\end{align}
